  ## Installation
  Install npm Node.js v14 or greater   
  nodemon: ^2.0.3  
  dotenv: ^8.2.0
  express: ^4.17.1
  mysql2: ^2.1.0
  sequelize: ^6.25.8  
  

  ## Usage
  "start": "node server.js"   
  "watch": "nodemon server.js"   
  "seed": "node seeds/index.js"
